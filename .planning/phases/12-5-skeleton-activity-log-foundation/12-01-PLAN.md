Создать `gui-app/src/shared/hooks/useActivityLog.ts`:

```typescript
import { invoke } from "@tauri-apps/api/core";
import { useCallback } from "react";

export type ActivityTag = "STATE" | "USER" | "PROTO" | "ERROR";

export function useActivityLog() {
  const log = useCallback(
    (tag: ActivityTag, message: string, details?: string) => {
      // Fire-and-forget - Activity Log не должен блокировать UI
      invoke("write_activity_log", { tag, message, details }).catch(() => {
        // Silent fail - logging failure must not affect UX
      });
    },
    []
  );

  return { log };
}
```

Ключевые решения:
- `ActivityTag` — строгий union type из 4 тегов per D-08: STATE, USER, PROTO, ERROR
- `useCallback` с пустым dependency array — стабильная ссылка
- Fire-and-forget через `.catch(() => {})` — Activity Log не должен блокировать UI flow
- Экспорт и type, и hook для потребителей
